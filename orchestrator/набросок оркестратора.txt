Фаза 1: "Архитектор" — Создание скелета проекта
Цель: Заставить LLM спроектировать структуру папок и файлов, а затем физически создать этот "скелет" на диске.

Напишите код для загрузки настроек: В orchestrator.py добавьте код для загрузки API-ключа и инициализации модели.

python
import os
import json
import google.generativeai as genai
from dotenv import load_dotenv

load_dotenv()
genai.configure(api_key=os.getenv("GEMINI_API_KEY"))
model = genai.GenerativeModel('gemini-pro')

def call_llm(prompt):
    """Универсальная функция для вызова LLM."""
    print("...Отправка запроса к LLM...")
    response = model.generate_content(prompt)
    # Простая очистка ответа
    return response.text.strip().replace("```json", "").replace("```", "").strip()

def create_project_skeleton(base_path, structure):
    """Рекурсивно создает папки и файлы из JSON-структуры."""
    for name, content in structure.items():
        path = os.path.join(base_path, name)
        if isinstance(content, dict): # Это папка
            print(f"Создание папки: {path}")
            os.makedirs(path, exist_ok=True)
            create_project_skeleton(path, content)
        else: # Это файл
            print(f"Создание файла: {path}")
            with open(path, 'w') as f:
                pass
Свяжите все вместе в основной функции:

python
def main():
    print("--- Фаза 1: Создание архитектуры проекта ---")
    with open("SPECIFICATION.md", 'r', encoding='utf-8') as f:
        spec_content = f.read()

    prompt = get_project_structure_prompt(spec_content)
    structure_json_str = call_llm(prompt)

    try:
        project_structure = json.loads(structure_json_str)
        create_project_skeleton(".", project_structure)
        print("--- Скелет проекта успешно создан! ---")
    except json.JSONDecodeError as e:
        print(f"Ошибка: LLM вернула невалидный JSON. {e}")
        print("Ответ от LLM:", structure_json_str)

if __name__ == "__main__":
    main()

# Функция для получения плоского списка файлов
def get_all_filepaths(structure, prefix=''):
    paths = []
    for name, content in structure.items():
        path = os.path.join(prefix, name)
        if isinstance(content, dict):
            paths.extend(get_all_filepaths(content, path))
        else:
            paths.append(path)
    return paths

all_files = get_all_filepaths(project_structure)

for file_path in all_files:
    print(f"Генерация кода для: {file_path}...")

    # Пропускаем файлы, которые не нужно генерировать (например, конфиги)
    if file_path.endswith(('.json', '.md')):
         print("Пропуск.")
         continue

    code_prompt = f"""
    На основе этой спецификации:
    ---
    {spec_content}
    ---
    Напиши полный, готовый к использованию код для файла `{file_path}`.
    Учитывай весь контекст проекта и его структуру.
    Верни ТОЛЬКО код, без каких-либо пояснений или markdown-оберток.
    """

    generated_code = call_llm(code_prompt)

    with open(file_path, 'w', encoding='utf-8') as f:

Фаза 3: "Сборщик/Тестировщик" — Итеративная отладка
Цель: Попытаться собрать проект и, если есть ошибки, отправить их обратно в LLM для исправления.

Напишите функцию для сборки:

python
import subprocess

def run_build():
    """Запускает команду сборки и возвращает результат."""
    print("\n--- Фаза 3: Попытка сборки проекта ---")
    # Сначала нужно установить зависимости
    subprocess.run(["npm", "install"], check=True, shell=True)

    # Затем запустить сборку
    result = subprocess.run(
        ["npm", "run", "tauri", "build"], 
        capture_output=True, 
        text=True, 
        shell=True
    )

    if result.returncode == 0:
        print("Сборка прошла успешно!")
        return None
    else:
        print("Сборка не удалась!")
        error_output = result.stdout + "\n" + result.stderr
        return error_output

# ... (дополняем main функцию)

max_retries = 5
for i in range(max_retries):
    error = run_build()
    if not error:
        print("Проект успешно собран!")
        break

    print(f"Попытка исправления #{i+1}...")

    debug_prompt = f"""
    Сборка проекта не удалась. Вот лог ошибки:
    ---
    {error}
    ---
    Проанализируй ошибку и предоставь исправленный код для файла, который ее вызвал.
    Вот полный текст спецификации для контекста:
    ---
    {spec_content}
    ---
    Верни ответ в формате JSON, где ключ - это полный путь к файлу, а значение - его полный, исправленный код.
    Например: {{"src/App.jsx": "const App = () => <div>Исправленный код</div>;"}}
    """

    fix_json_str = call_llm(debug_prompt)

    try:
        fixes = json.loads(fix_json_str)
        for file_path, corrected_code in fixes.items():
            print(f"Применение исправления для {file_path}...")
            with open(file_path, 'w', encoding='utf-8') as f:
                f.write(corrected_code)
    except json.JSONDecodeError:
        print("Не удалось исправить ошибку, LLM вернула невалидный JSON.")
        break
else:
    print("Не удалось собрать проект после нескольких попыток.")
